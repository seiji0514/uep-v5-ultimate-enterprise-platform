#### 実装方法

def load_interview_preparation_data(company_name: str):
    """
    既存の面談準備資料から情報を読み込む

            # 既存資料から抽出
        }
        
        # 質問事項リストを読み込み

            "government-ai-security-platform/AIexe-質問事項リスト_面談用最終版.md"

            "company_info": company_info,

### 方法2: 質問事項リストの活用

#### 実装方法

既存の質問事項リストを読み込んで、面談練習に活用：

```python
def load_question_list(file_path: str):
    """
    質問事項リストを読み込む
    """
    with open(file_path, 'r', encoding='utf-8') as f:
        content = f.read()
    
    # Markdownから質問を抽出
    questions = []
    for line in content.split('\n'):
        if line.startswith('-') or line.startswith('*'):
            # 質問項目を抽出
            question = line.strip('- *').strip()
            if '?' in question or '？' in question:
                questions.append(question)
    
    return questions
```

#### 使用例

```python
# 質問事項リストを読み込む
questions = load_question_list(
    "government-ai-security-platform/AIexe-質問事項リスト_面談用最終版.md"
)

# 面談練習モードで使用
for question in questions:
    # AIが応答を生成
    response = await interview_service.generate_interview_response(
        question, connection_id, interview_type="general"
    )
    print(f"Q: {question}")
    print(f"A: {response}\n")
```

### 方法4: 業界用語集の活用

#### 実装方法

業界用語集を読み込んで、専門用語の説明を強化：

```python
def load_industry_terms(file_path: str):
    """
    業界用語集を読み込む
    """
    terms = {}
    with open(file_path, 'r', encoding='utf-8') as f:
        content = f.read()
    
    # CSVまたはMarkdownから用語を抽出
    # 例: {"LLM": "大規模言語モデル", "RAG": "Retrieval-Augmented Generation", ...}
    
    return terms
```

#### 使用例

```python
# 業界用語集を読み込む
terms = load_industry_terms(
    "government-ai-security-platform/AIexe-面談用_業界用語集.md"
)

# 専門用語が含まれる質問に対して、用語の説明を追加
def enhance_response_with_terms(response: str, question: str):
    for term, definition in terms.items():
        if term in question:
            response += f"\n（補足: {term}は{definition}のことです）"
    return response
```

## 🚀 統合システムの実装

### 統合サービスクラス

```python
class IntegratedInterviewService:
    """
    既存システムと統合した面談支援サービス
    """
    
    def __init__(self):
        self.interview_service = InterviewResponseService()
        self.preparation_data = {}
        self.question_lists = {}
        self.demo_data = {}
        self.industry_terms = {}
    
    def load_all_preparation_data(self, company_name: str):
        """すべての準備資料を読み込む"""
        # 企業情報
        self.preparation_data = load_interview_preparation_data(company_name)
        
        # 質問事項リスト
        self.question_lists = load_question_list(
            f"government-ai-security-platform/{company_name}-質問事項リスト_面談用最終版.md"
        )
        
        # デモンストレーション準備
        self.demo_data = load_demo_preparation(
            f"government-ai-security-platform/{company_name}-デモンストレーション準備ガイド.md"
        )
        
        # 業界用語集
        self.industry_terms = load_industry_terms(
            f"government-ai-security-platform/{company_name}-面談用_業界用語集.md"
        )
        
        # 統合情報を設定
        self.interview_service.set_all_info(
            candidate_info=self._build_candidate_info(),
            company_info=self.preparation_data["company_info"],
            interviewer_info=self.preparation_data.get("interviewer_info", {})
        )
    
    def practice_mode(self, question: str):
        """面談練習モード"""
        # 業界用語を考慮した応答生成
        response = await self.interview_service.generate_interview_response(
            question, connection_id=0, interview_type="general"
        )
        
        # 業界用語の説明を追加
        response = enhance_response_with_terms(response, question)
        
        return response
    
    def _build_candidate_info(self):
        """デモンストレーション準備から候補者情報を構築"""
        return {
            "skills": self.demo_data.get("tech_stack", []),
            "experience": format_achievements(self.demo_data.get("achievements", [])),
            # その他の情報
        }
```

## 📋 実用的な使用フロー

### フロー1: 面談準備（事前）

```
1. 企業名を指定して準備資料を読み込む
   → load_all_preparation_data("AIexe")

2. 質問事項リストで練習
   → practice_mode("自己紹介をお願いします")

3. AIが応答を生成し、改善点を提案
   → 応答を確認して改善

4. 実際の面談に臨む
```

### フロー2: リアルタイム支援（面談中）

```
1. 面談開始前にシステムを起動
   → 準備資料を読み込む

2. 質問を音声認識
   → リアルタイムで文字起こし

3. AIが応答を生成（既存資料を考慮）
   → 画面に表示

4. ユーザーが確認してから回答
```

### フロー3: 面談後の振り返り

```
1. 面談の音声を録音
   → システムで文字起こし

2. 質問と応答を分析
   → 既存の質問事項リストと比較

3. 改善点を提案
   → 次回の面談準備に活用
```
